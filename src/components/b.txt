class NotificationService {
  constructor() {
    this.audioContext = null;
    this.audioBuffer = null;
    this.enabled = true;
    this.audioReady = false;
  }

  async enableAudio() {
    try {
      // Táº¡o AudioContext
      const AudioContext = window.AudioContext || window.webkitAudioContext;
      this.audioContext = new AudioContext();
      
      // Fetch vÃ  decode audio file
      const response = await fetch('/sounds/notification.wav');
      const arrayBuffer = await response.arrayBuffer();
      this.audioBuffer = await this.audioContext.decodeAudioData(arrayBuffer);
      
      this.audioReady = true;
      console.log('âœ“ Audio context ready');
      
      return true;
    } catch (error) {
      console.error('Error enabling audio:', error);
      throw error;
    }
  }

  async playSound() {
    if (!this.enabled || !this.audioReady || !this.audioContext || !this.audioBuffer) {
      console.log('Sound not ready');
      return;
    }
    
    try {
      // Resume context náº¿u bá»‹ suspend
      if (this.audioContext.state === 'suspended') {
        await this.audioContext.resume();
      }
      
      // Táº¡o source má»›i má»—i láº§n play
      const source = this.audioContext.createBufferSource();
      source.buffer = this.audioBuffer;
      
      // Táº¡o gain node Ä‘á»ƒ Ä‘iá»u chá»‰nh volume
      const gainNode = this.audioContext.createGain();
      gainNode.gain.value = 0.7;
      
      source.connect(gainNode);
      gainNode.connect(this.audioContext.destination);
      
      source.start(0);
      console.log('âœ“ Sound played');
    } catch (error) {
      console.error('Sound play error:', error);
    }
  }

  showNotification(title, body) {
    if (!this.enabled) return;
    
    if ('Notification' in window && Notification.permission === 'granted') {
      const notification = new Notification(title, {
        body: body,
        icon: '/logo192.png',
        badge: '/logo192.png',
        tag: 'new-message',
        requireInteraction: false,
        silent: false // CÃ³ Ã¢m thanh há»‡ thá»‘ng
      });
      
      setTimeout(() => notification.close(), 5000);
      
      notification.onclick = () => {
        window.focus();
        notification.close();
      };
    }
  }

  notify(customerName, message) {
    console.log('ðŸ”” Notification triggered:', customerName);
    this.playSound();
    this.showNotification(
      `ðŸ’¬ ${customerName}`,
      message.substring(0, 100)
    );
  }

  setEnabled(enabled) {
    this.enabled = enabled;
  }

  static async init() {
    if (!("Notification" in window)) {
      console.log("This browser does not support notifications");
      return;
    }

    if (Notification.permission === "default") {
      await Notification.requestPermission();
    }
  }

  static notify(title, body) {
    if (!("Notification" in window)) return;
    
    if (Notification.permission === "granted") {
      new Notification(title, { 
        body,
        icon: '/logo192.png' // Optional: add your app icon
      });
      
      // Play sound
      const audio = new Audio('/notification.mp3');
      audio.play().catch(e => console.log('Audio play failed:', e));
    }
  }
}

const notificationService = new NotificationService();
export default notificationService;
